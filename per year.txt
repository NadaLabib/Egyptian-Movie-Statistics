def checkIfYear(y):
    if len(y)!= 4:
        return False
    for n in y :
        if not n.isdigit():
            return False
    return True

year = []
def takingYears(l):
    for i in range(len(l)):
        for j in range(len(l[i])):
            if checkIfYear(l[i][j]):
                year.append(l[i][j])
    return year
