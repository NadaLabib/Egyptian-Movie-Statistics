def checkIfMonth(m):
    if m == '':
        return False
    for n in m :
        if n.isdigit():
            return False
    return True

month = []
def takingMonth(l):
    for i in range(len(l)):
        for j in range(len(l[i])):
            if checkIfMonth(l[i][j]):
                month.append(l[i][j])
    return month

def countingMonths (o):
    for i in range(len(o)):

        for j in range(len(o)):
